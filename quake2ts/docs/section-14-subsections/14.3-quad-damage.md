int P_DamageModifier(edict_t *ent) {
  int damage_modifier = 1;
  if (ent->client->quad_time > level.time)
    damage_modifier *= 4;
  if (ent->client->double_time > level.time)
    damage_modifier *= 2;
  return damage_modifier;
}
